--Section A.

CREATE TABLE Author (
  AuthorID INT PRIMARY KEY,
  AuthorFirstName VARCHAR(10),
  AuthorLastName VARCHAR(10),
  AuthorNationality VARCHAR(20)
);
--Creates the Author table with AuthorID as the primary key


CREATE TABLE Book (
  BookID INT PRIMARY KEY,
  BookTitle VARCHAR(30),
  BookAuthor INT,
  Genre VARCHAR(10),
  FOREIGN KEY (BookAuthor) REFERENCES Author(AuthorID)
);
--Creates the Book table with BookID and BookAuthor as a foreign key to the Author table


CREATE TABLE Client (
  ClientID INT PRIMARY KEY,
  ClientFirstName VARCHAR(10),
  ClientLastName VARCHAR(10),
  ClientDOB INT,
  Occupation VARCHAR(30)
);
--Creates the Client table with ClientID as the primary key


CREATE TABLE Borrower (
  BorrowID INT PRIMARY KEY,
  ClientID INT,
  BookID INT,
  BorrowDate DATE,
  FOREIGN KEY (ClientID) REFERENCES Client(ClientID),
  FOREIGN KEY (BookID) REFERENCES Book(BookID)
);
--Creates the Borrower table with BorrowID as the primary key, and both ClientID and BookID as foreign keys to the Client and Book tables respectively
